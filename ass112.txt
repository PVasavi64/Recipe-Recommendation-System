function searchRecipe() {

  let recipe =
    document.getElementById("searchBox").value.toLowerCase().trim();

  let content =
    document.getElementById("recipeContent");

  if (recipe === "pasta" || recipe === "creamy pasta") {

    content.innerHTML = `
      <h2>Recipe: Creamy Pasta</h2>

      <img src="pas.jpg" alt="Creamy Pasta" class="image">

      <h3>Ingredients</h3>
      <ul>
        <li>200g pasta</li>
        <li>1 cup cream</li>
        <li>2 cloves garlic (chopped)</li>
        <li>1/4 cup grated cheese</li>
        <li>Salt & pepper to taste</li>
      </ul>

      <h3>Instructions</h3>
      <ol>
        <li>Boil the pasta until cooked.</li>
        <li>In a pan, sauté garlic in butter.</li>
        <li>Add cream and bring to a simmer.</li>
        <li>Add cheese, salt, and pepper.</li>
        <li>Mix pasta into the sauce and serve hot.</li>
      </ol>
    `;
  }

  else if (recipe === "pizza") {

    content.innerHTML = `
      <h2>Recipe: Pizza</h2>

      <h3>Ingredients</h3>
      <ul>
        <li>Pizza Base</li>
        <li>Cheese</li>
        <li>Tomato Sauce</li>
        <li>Vegetables</li>
      </ul>

      <h3>Instructions</h3>
      <ol>
        <li>Spread sauce on pizza base.</li>
        <li>Add vegetables and cheese.</li>
        <li>Bake for 15 minutes.</li>
        <li>Serve hot.</li>
      </ol>
    `;
  }

  else if (recipe === "biryani") {

    content.innerHTML = `
      <h2>Recipe: Biryani</h2>

      <h3>Ingredients</h3>
      <ul>
        <li>Rice</li>
        <li>Chicken</li>
        <li>Spices</li>
        <li>Onions</li>
      </ul>

      <h3>Instructions</h3>
      <ol>
        <li>Cook chicken with spices.</li>
        <li>Prepare rice separately.</li>
        <li>Layer rice and chicken.</li>
        <li>Cook on low flame.</li>
        <li>Serve hot.</li>
      </ol>
    `;
  }

  else {

    content.innerHTML = `
      <h2>Recipe Not Found</h2>
      <p>Try searching for:</p>
      <ul>
        <li>Pasta</li>
        <li>Pizza</li>
        <li>Biryani</li>
      </ul>
    `;
  }
}
